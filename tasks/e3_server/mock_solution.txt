class ItemStore:
    def __init__(self):
        self._items = {}
        self._next = 1

    def add(self, name, price):
        if not name:
            raise ValueError('name required')
        if price < 0:
            raise ValueError('price must be >= 0')
        item_id = self._next
        self._next += 1
        self._items[item_id] = {'id': item_id, 'name': name, 'price': price}
        return item_id

    def get(self, item_id):
        return self._items[item_id]

    def filter_by_max_price(self, limit):
        return [self._items[k] for k in sorted(self._items)
                if self._items[k]['price'] <= limit]
