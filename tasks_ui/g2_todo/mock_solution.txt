<!doctype html>
<html lang="ja"><head><meta charset="utf-8"><style>
  body { font-family: sans-serif; max-width: 400px; margin: 40px auto; }
  li { cursor: pointer; padding: 6px; }
  li.done { text-decoration: line-through; color: #999; }
</style></head><body>
  <h1>TODO</h1>
  <input type="text" id="t" placeholder="やること">
  <button onclick="add()">追加</button>
  <div id="count">残り 0 件</div>
  <ul id="list"></ul>
  <script>
    const list = document.getElementById('list');
    const t = document.getElementById('t');
    function render() {
      const remain = list.querySelectorAll('li:not(.done)').length;
      document.getElementById('count').textContent = '残り ' + remain + ' 件';
    }
    function add() {
      if (!t.value) return;
      const li = document.createElement('li');
      li.textContent = t.value;
      li.onclick = () => { li.classList.toggle('done'); render(); };
      list.appendChild(li);
      t.value = '';
      render();
    }
  </script>
</body></html>
